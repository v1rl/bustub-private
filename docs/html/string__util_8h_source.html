41  template <typename C, typename S, typename Func>
42  static auto Join(const C &input, S count, const std::string &separator, Func f) -> std::string {
43  // The result
44  std::string result;
45 
46  // If the input isn't empty, append the first element. We do this so we
47  // don't need to introduce an if into the loop.
48  if (count > 0) {
49  result += f(input[0]);
50  }
51 
52  // Append the remaining input components, after the first
53  for (size_t i = 1; i < count; i++) {
54  result += separator + f(input[i]);
55  }
56 
57  return result;
58  }
