139  auto CreateTable(Transaction *txn, const std::string &table_name, const Schema &schema, bool create_table_heap = true)
140  -> std::shared_ptr<TableInfo> {
141  if (table_names_.count(table_name) != 0) {
142  return NULL_TABLE_INFO;
143  }
144 
145  // Construct the table heap
146  std::unique_ptr<TableHeap> table = nullptr;
147 
148  // When create_table_heap == false, it means that we're running binder tests (where no txn will be provided) or
149  // we are running shell without buffer pool. We don't need to create TableHeap in this case.
150  if (create_table_heap) {
151  table = std::make_unique<TableHeap>(bpm_);
152  } else {
153  // Otherwise, create an empty heap only for binder tests
154  table = TableHeap::CreateEmptyHeap(create_table_heap);
155  }
156 
157  // Fetch the table OID for the new table
158  const auto table_oid = next_table_oid_.fetch_add(1);
159 
160  // Construct the table information
161  auto meta = std::make_shared<TableInfo>(schema, table_name, std::move(table), table_oid);
162 
163  // Update the internal tracking mechanisms
164  tables_.emplace(table_oid, meta);
165  table_names_.emplace(table_name, table_oid);
166  index_names_.emplace(table_name, std::unordered_map<std::string, index_oid_t>{});
167 
168  return meta;
169  }

215  template <class KeyType, class ValueType, class KeyComparator>
216  auto CreateIndex(Transaction *txn, const std::string &index_name, const std::string &table_name, const Schema &schema,
217  const Schema &key_schema, const std::vector<uint32_t> &key_attrs, std::size_t keysize,
218  HashFunction<KeyType> hash_function, bool is_primary_key = false,
219  IndexType index_type = IndexType::BPlusTreeIndex) -> std::shared_ptr<IndexInfo> {
220  // Reject the creation request for nonexistent table
221  if (table_names_.find(table_name) == table_names_.end()) {
222  return NULL_INDEX_INFO;
223  }
224 
225  // If the table exists, an entry for the table should already be present in index_names_
226  BUSTUB_ASSERT((index_names_.find(table_name) != index_names_.end()), "Broken Invariant");
227 
228  // Determine if the requested index already exists for this table
229  auto &table_indexes = index_names_.find(table_name)->second;
230  if (table_indexes.find(index_name) != table_indexes.end()) {
231  // The requested index already exists for this table
232  return NULL_INDEX_INFO;
233  }
234 
235  // Construct index metdata
236  auto meta = std::make_unique<IndexMetadata>(index_name, table_name, &schema, key_attrs, is_primary_key);
237 
238  // Construct the index, take ownership of metadata
239  // TODO(Kyle): We should update the API for CreateIndex
240  // to allow specification of the index type itself, not
241  // just the key, value, and comparator types
242 
243  // TODO(chi): support both hash index and btree index
244  std::unique_ptr<Index> index;
245  if (index_type == IndexType::HashTableIndex) {
246  index = std::make_unique<ExtendibleHashTableIndex<KeyType, ValueType, KeyComparator>>(std::move(meta), bpm_,
247  hash_function);
248  } else if (index_type == IndexType::BPlusTreeIndex) {
249  index = std::make_unique<BPlusTreeIndex<KeyType, ValueType, KeyComparator>>(std::move(meta), bpm_);
250  } else if (index_type == IndexType::STLOrderedIndex) {
251  index = std::make_unique<STLOrderedIndex<KeyType, ValueType, KeyComparator>>(std::move(meta), bpm_);
252  } else if (index_type == IndexType::STLUnorderedIndex) {
253  index =
254  std::make_unique<STLUnorderedIndex<KeyType, ValueType, KeyComparator>>(std::move(meta), bpm_, hash_function);
255  } else {
256  UNIMPLEMENTED("Unsupported Index Type");
257  }
258 
259  // Populate the index with all tuples in table heap
260  auto table_meta = GetTable(table_name);
261  for (auto iter = table_meta->table_->MakeIterator(); !iter.IsEnd(); ++iter) {
262  auto [meta, tuple] = iter.GetTuple();
263  // we have to silently ignore the error here for a lot of reasons...
264  index->InsertEntry(tuple.KeyFromTuple(schema, key_schema, key_attrs), tuple.GetRid(), txn);
265  }
266 
267  // Get the next OID for the new index
268  const auto index_oid = next_index_oid_.fetch_add(1);
269 
270  // Construct index information; IndexInfo takes ownership of the Index itself
271  auto index_info = std::make_shared<IndexInfo>(key_schema, index_name, std::move(index), index_oid, table_name,
272  keysize, is_primary_key, index_type);
273  // Update internal tracking
274  indexes_.emplace(index_oid, index_info);
275  table_indexes.emplace(index_name, index_oid);
276  return index_info;
277  }

342  auto GetTableIndexes(const std::string &table_name) const -> std::vector<std::shared_ptr<IndexInfo>> {
343  // Ensure the table exists
344  if (table_names_.find(table_name) == table_names_.end()) {
345  return std::vector<std::shared_ptr<IndexInfo>>{};
346  }
347 
348  auto table_indexes = index_names_.find(table_name);
349  BUSTUB_ASSERT((table_indexes != index_names_.end()), "Broken Invariant");
350 
351  std::vector<std::shared_ptr<IndexInfo>> indexes{};
352  indexes.reserve(table_indexes->second.size());
353  for (const auto &index_meta : table_indexes->second) {
354  auto index = indexes_.find(index_meta.second);
355  BUSTUB_ASSERT((index != indexes_.end()), "Broken Invariant");
356  indexes.push_back(index->second);
357  }
358 
359  return indexes;
360  }
