23 // This class is used to guard the value returned by the trie. It holds a reference to the root so
24 // that the reference to the value will not be invalidated.
25 template <class T>
26 class ValueGuard {
27  public:
28  ValueGuard(Trie root, const T &value) : root_(std::move(root)), value_(value) {}
29  auto operator*() const -> const T & { return value_; }
30 
31  private:
32  Trie root_;
33  const T &value_;
34 };
