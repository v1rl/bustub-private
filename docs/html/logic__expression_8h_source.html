73  auto GetBoolAsCmpBool(const Value &val) const -> CmpBool {
74  if (val.IsNull()) {
75  return CmpBool::CmpNull;
76  }

83  auto PerformComputation(const Value &lhs, const Value &rhs) const -> CmpBool {
84  auto l = GetBoolAsCmpBool(lhs);
85  auto r = GetBoolAsCmpBool(rhs);
86  switch (logic_type_) {
87  case LogicType::And:
88  if (l == CmpBool::CmpFalse || r == CmpBool::CmpFalse) {
89  return CmpBool::CmpFalse;
90  }
91  if (l == CmpBool::CmpTrue && r == CmpBool::CmpTrue) {
92  return CmpBool::CmpTrue;
93  }
94  return CmpBool::CmpNull;
95  case LogicType::Or:
96  if (l == CmpBool::CmpFalse && r == CmpBool::CmpFalse) {
97  return CmpBool::CmpFalse;
98  }
99  if (l == CmpBool::CmpTrue || r == CmpBool::CmpTrue) {
100  return CmpBool::CmpTrue;
101  }
102  return CmpBool::CmpNull;
103  default:
104  UNREACHABLE("Unsupported logic type.");
105  }
106  }
