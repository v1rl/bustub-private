45  // Comparison functions
46  //
47  // NOTE:
48  // We could get away with only CompareLessThan() being purely virtual, since
49  // the remaining comparison functions can derive their logic from
50  // CompareLessThan(). For example:
51  //
52  // CompareEquals(o) = !CompareLessThan(o) && !o.CompareLessThan(this)
53  // CompareNotEquals(o) = !CompareEquals(o)
54  // CompareLessThanEquals(o) = CompareLessThan(o) || CompareEquals(o)
55  // CompareGreaterThan(o) = !CompareLessThanEquals(o)
56  // ... etc. ...
57  //
58  // We don't do this for two reasons:
59  // (1) The redundant calls to CompareLessThan() may be a performance problem,
60  // and since Value is a core component of the execution engine, we want to
61  // make it as performant as possible.
62  // (2) Keep the interface consistent by making all functions purely virtual.
63  virtual auto CompareEquals(const Value &left, const Value &right) const -> CmpBool;
64  virtual auto CompareNotEquals(const Value &left, const Value &right) const -> CmpBool;
65  virtual auto CompareLessThan(const Value &left, const Value &right) const -> CmpBool;
66  virtual auto CompareLessThanEquals(const Value &left, const Value &right) const -> CmpBool;
