138  static inline auto CastAsBigInt(const Value &value) -> Value {
139  if (Type::GetInstance(TypeId::BIGINT)->IsCoercableFrom(value.GetTypeId())) {
140  if (value.IsNull()) {
141  return ValueFactory::GetBigIntValue(static_cast<int64_t>(BUSTUB_INT64_NULL));
142  }
143  switch (value.GetTypeId()) {
144  case TypeId::TINYINT:
145  return ValueFactory::GetBigIntValue(static_cast<int64_t>(value.GetAs<int8_t>()));
146  case TypeId::SMALLINT:
147  return ValueFactory::GetBigIntValue(static_cast<int64_t>(value.GetAs<int16_t>()));
148  case TypeId::INTEGER:
149  return ValueFactory::GetBigIntValue(static_cast<int64_t>(value.GetAs<int32_t>()));
150  case TypeId::BIGINT:
151  return ValueFactory::GetBigIntValue(value.GetAs<int64_t>());
152  case TypeId::DECIMAL: {
153  if (value.GetAs<double>() > static_cast<double>(BUSTUB_INT64_MAX) ||
154  value.GetAs<double>() < static_cast<double>(BUSTUB_INT64_MIN)) {
155  throw Exception(ExceptionType::OUT_OF_RANGE, "Numeric value out of range.");
156  }
157  return ValueFactory::GetBigIntValue(static_cast<int64_t>(value.GetAs<double>()));
158  }
159  case TypeId::VARCHAR: {
160  std::string str = value.ToString();
161  int64_t bigint = 0;
162  try {
163  bigint = stoll(str);
164  } catch (std::out_of_range &e) {
165  throw Exception(ExceptionType::OUT_OF_RANGE, "Numeric value out of range.");
166  } catch (std::invalid_argument &e) {
167  throw Exception("Invalid input syntax for bigint: \'" + str + "\'");
168  }
169  if (bigint > BUSTUB_INT64_MAX || bigint < BUSTUB_INT64_MIN) {
170  throw Exception(ExceptionType::OUT_OF_RANGE, "Numeric value out of range.");
171  }
172  return ValueFactory::GetBigIntValue(bigint);
173  }
174  default:
175  break;
176  }
177  }
178  throw Exception(Type::GetInstance(value.GetTypeId())->ToString(value) + " is not coercable to BIGINT.");

400  static inline auto CastAsTimestamp(const Value &value) -> Value {
401  if (Type::GetInstance(TypeId::TIMESTAMP)->IsCoercableFrom(value.GetTypeId())) {
402  if (value.IsNull()) {
403  return ValueFactory::GetTimestampValue(BUSTUB_TIMESTAMP_NULL);
404  }
405  switch (value.GetTypeId()) {
406  case TypeId::TIMESTAMP:
407  return ValueFactory::GetTimestampValue(value.GetAs<uint64_t>());
408  case TypeId::VARCHAR: {
409  std::string str = value.ToString();
410  if (str.length() == 22) {
411  str = str.substr(0, 19) + ".000000" + str.substr(19, 3);
412  }
413  if (str.length() != 29) {
414  throw Exception("Timestamp format error.");
415  }
416  if (str[10] != ' ' || str[4] != '-' || str[7] != '-' || str[13] != ':' || str[16] != ':' || str[19] != '.' ||
417  (str[26] != '+' && str[26] != '-')) {
418  throw Exception("Timestamp format error.");
419  }
420  bool is_digit[29] = {true, true, true, true, false, true, true, false, true, true,
421  false, true, true, false, true, true, false, true, true, false,
422  true, true, true, true, true, true, false, true, true};
423  for (int i = 0; i < 29; i++) {
424  if (is_digit[i]) {
425  if (str[i] < '0' || str[i] > '9') {
426  throw Exception("Timestamp format error.");
427  }
428  }
429  }
430  int tz = 0;
431  uint32_t year = 0;
432  uint32_t month = 0;
433  uint32_t day = 0;
434  uint32_t hour = 0;
435  uint32_t min = 0;
436  uint32_t sec = 0;
437  uint32_t micro = 0;
438  uint64_t res = 0;
439  if (sscanf(str.c_str(), "%4u-%2u-%2u %2u:%2u:%2u.%6u%3d", &year, &month, &day, &hour, &min, &sec, &micro,
440  &tz) != 8) {
441  throw Exception("Timestamp format error.");
442  }
443  if (year > 9999 || month > 12 || day > 31 || hour > 23 || min > 59 || sec > 59 || micro > 999999 ||
444  day == 0 || month == 0) {
445  throw Exception(ExceptionType::OUT_OF_RANGE, "Timestamp value out of range.");
446  }
447  uint32_t max_day[13] = {0, 31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31};
448  uint32_t max_day_lunar[13] = {0, 31, 29, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31};
449  if ((year % 4 == 0 && year % 100 != 0) || year % 400 == 0) {
450  if (day > max_day_lunar[month]) {
451  throw Exception(ExceptionType::OUT_OF_RANGE, "Timestamp value out of range.");
452  }
453  } else if (day > max_day[month]) {
454  throw Exception(ExceptionType::OUT_OF_RANGE, "Timestamp value out of range.");
455  }
456  uint32_t timezone = tz + 12;
457  if (tz > 26) {
458  throw Exception("Timestamp format error.");
459  }
460  res += month;
461  res *= 32;
462  res += day;
463  res *= 27;
464  res += timezone;
465  res *= 10000;
466  res += year;
467  res *= 100000;
468  res += hour * 3600 + min * 60 + sec;
469  res *= 1000000;
470  res += micro;
471  return ValueFactory::GetTimestampValue(res);
472  }
473  default:
474  break;
475  }
476  }
477  throw Exception(Type::GetInstance(value.GetTypeId())->ToString(value) + " is not coercable to TIMESTAMP.");
478  }
